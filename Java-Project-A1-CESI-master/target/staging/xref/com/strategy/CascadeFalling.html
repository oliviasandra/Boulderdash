11  /**
12   * The CascadeFalling class.
13   * @author Antoine Chauvel
14   * @version 1.0
15   *
16   */
17  public class CascadeFalling extends Strategy<MobileElements> {
18  
19  	
20  	/**
21  	 * The CascadeFalling strategy constructor.
22  	 * @param me the mobile element concerned
23  	 */
24  	public CascadeFalling(MobileElements me) {
25  		this.me = me;
26  	}
27  
28  	@Override
29  	public void runStrategy() {
30  		
31  		Entity getNextEntity = me.getMap().getArrayMap()[me.getPositionX()][me.getPositionY()+1];
32  		Entity getLeftEntity = me.getMap().getArrayMap()[me.getPositionX() - 1][me.getPositionY()];
33  		Entity getLeftBottomEntity = me.getMap().getArrayMap()[me.getPositionX() - 1][me.getPositionY() +1];
34  		Entity getRightEntity = me.getMap().getArrayMap()[me.getPositionX() + 1][me.getPositionY()];
35  		Entity getRightBottomEntity = me.getMap().getArrayMap()[me.getPositionX() + 1][me.getPositionY() +1];
36  		final int bonusEnemyKilled = 4;
37  		
38  		if (getNextEntity instanceof Path) {
39  			this.cascadeFalling(0, 1);
40  		} else if (getNextEntity instanceof Player && me.getIsFallen()){
41  			((Player)getNextEntity).setIsAlive(false);
42  			
43  		} else if(getNextEntity instanceof Enemy && me.getIsFallen()) {
44  			((Enemy)getNextEntity).setIsAlive(false);
45  			me.getMap().getArrayMap()[me.getPositionX()][me.getPositionY()+1] = new Path(me.getPositionX(), me.getPositionY());
46  			me.getMap().getPlayer().increaseDiamondsCounter(bonusEnemyKilled);
47  		} else if(getNextEntity instanceof Stone && getLeftEntity instanceof Path && getLeftBottomEntity instanceof Path) {	
48  			this.cascadeFalling(-1, 1);
49  		 
50  		} else if(getNextEntity instanceof Stone && getRightEntity instanceof Path && getRightBottomEntity instanceof Path) {	
51  			this.cascadeFalling(1, 1);
52  		} else {
53  			me.setIsFallen(false);
54  		}
55  	}
56  	
57  	
58  	
59  	/**
60  	 * The cascadeFalling method.
61  	 * Defines the falling of the mobile element.
62  	 * @param sideX the x side on which the entity is going to fell
63  	 * @param sideY the y side on which the entity is going to fell
64  	 */
65  	public void cascadeFalling(int sideX, int sideY) {
66  		
67  		me.getMap().getArrayMap()[me.getPositionX() + sideX][me.getPositionY() + sideY] = me;
68  		me.getMap().getArrayMap()[me.getPositionX()][me.getPositionY()] = new Path(me.getPositionX(), me.getPositionY());
69  		me.setIsFallen(true);
70  		me.setPositionX(me.getPositionX()+ sideX);
71  		me.setPositionY(me.getPositionY()+ sideY);
72  		
73  	}
74  
75  	@Override
76  	public String returnStrategy() {
77  		return "CascadeFalling";
78  	}
79  	
80  
81  }
